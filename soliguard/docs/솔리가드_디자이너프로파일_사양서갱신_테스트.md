# 솔리가드(SoliGuard) 디자이너 프로파일 · 사양서 갱신 · 디자인 파일 테스트

디자이너 직무 프로파일 보강, 제품 사양서 업데이트 항목, PSD/XD/Figma 검사 단위 테스트를 정리한 문서입니다.

---

## 1. 디자이너 직무 프로파일 & 사양서 업데이트

### 1-1. `scanner.py` 직무 프로파일 업데이트

디자이너 프로파일이 디자인 파일을 우선 스캔하도록 대상 확장자와 적용 규칙을 보강합니다.

```python
# src/soliguard/scanner.py (직무 프로파일 부분 업데이트)
from soliguard.detectors import RULES, SECRET_RULES

# 직무별 적용 검출 규칙
PROFILE_RULES = {
    "개발자": RULES + SECRET_RULES,
    "디자이너": RULES,            # 디자인 파일 텍스트/OCR 결과에 동일 적용
    "기획자": RULES,
    "PM": RULES,
    "전산사무": RULES,
}

# 직무별 우선 스캔 대상 확장자 (스캔 설정 기본값 구성에 사용)
PROFILE_EXTENSIONS = {
    "개발자": {".py", ".java", ".js", ".sql", ".yml", ".yaml",
              ".env", ".properties", ".json", ".log",
              ".txt", ".csv", ".xlsx", ".docx", ".pdf"},
    "디자이너": {".psd", ".psb", ".xd",          # 디자인 원본
                ".jpg", ".jpeg", ".png", ".bmp", ".tiff",  # 시안·리소스 이미지
                ".pdf", ".docx", ".xlsx"},
    "기획자": {".docx", ".hwp", ".hwpx", ".xlsx", ".csv", ".pdf", ".txt"},
    "PM": {".docx", ".hwp", ".hwpx", ".xlsx", ".csv", ".pdf", ".txt"},
    "전산사무": {".xlsx", ".csv", ".docx", ".hwp", ".hwpx", ".pdf", ".txt"},
}

# 직무별 권장 스캔 폴더(온보딩 기본 추천)
PROFILE_FOLDERS = {
    "개발자": ["Downloads", "Desktop", "Projects", "workspace", "git"],
    "디자이너": ["Downloads", "Desktop", "Documents", "Design", "시안", "작업"],
    "기획자": ["Downloads", "Desktop", "Documents", "산출물"],
    "PM": ["Downloads", "Desktop", "Documents", "프로젝트"],
    "전산사무": ["Downloads", "Desktop", "Documents"],
}

# 디자이너 프로파일은 OCR을 기본 권장(이미지로 합쳐진 개인정보 대응)
PROFILE_OCR_DEFAULT = {
    "개발자": False,
    "디자이너": True,
    "기획자": False,
    "PM": False,
    "전산사무": False,
}
```

이렇게 하면 온보딩에서 "디자이너"를 선택했을 때 디자인 폴더와 `.psd`·`.xd` 확장자가 미리 체크되고 OCR이 기본 켜진 상태로 구성됩니다. Figma는 클라우드 옵트인 기능이므로 직무 프로파일의 기본 대상에는 넣지 않고, 디자이너 사용자에게만 설정에서 "Figma 검사(고급)"를 권유 노출합니다.

### 1-2. 제품 사양서 업데이트 항목

사양서의 해당 항목을 다음과 같이 갱신합니다.

#### 4. 핵심 기능 명세 — 지원 포맷 (갱신)

지원 포맷에 디자인 파일을 추가한다. 기존 txt·csv·log·소스코드·xlsx·xls·docx·hwp·hwpx·pdf·이미지(OCR)에 더해, **PSD/PSB**(Photoshop, 텍스트 레이어 직접 추출 + 래스터 레이어 OCR 폴백)와 **XD**(Adobe XD, ZIP 내부 JSON 텍스트 노드 추출)를 지원한다. Figma는 클라우드 저장이 기본이므로 별도의 옵트인 고급 기능으로 분리한다.

#### 4. 핵심 기능 명세 — 검출 대상 표에 행 추가

| 정보 유형 | 1차 탐지 | 2차 검증 | 위험도 | 대상 직무 |
|---|---|---|---|---|
| 디자인 텍스트 레이어 속 정보 | PSD/XD 텍스트 추출 후 기존 검출 로직 | 각 유형별 검증 | 유형 따름 | 디자이너 |
| 시안 내 이미지 속 정보 | 래스터 레이어 OCR 후 검출 | — | 높음 | 디자이너 |

#### 5. 차별화 전략 (보강)

디자인 파일(PSD·XD) 텍스트 레이어를 OCR 없이 정확히 검출하는 기능은 일반 개인정보 도구는 물론 기존 SI 보안 솔루션에도 드문 기능으로, 디자이너 직무를 실질적으로 지원하는 차별점을 더한다. 디자이너가 시안·목업에 잘못 삽입한 실고객 정보(텍스트 또는 신분증 이미지)를 잡아낸다.

#### 10. 보안·컴플라이언스 (Figma 항목 추가)

Figma 클라우드 검사는 솔리가드의 "로컬 처리·외부 미전송" 원칙과 충돌하므로 기본 비활성의 옵트인 기능으로만 제공한다. 사용자의 명시적 동의와 액세스 토큰이 모두 있을 때만 동작하며, Figma 서버에서 텍스트 노드만 가져와 메모리에서 검출에 사용한 뒤 즉시 폐기한다. 이미지·디자인 원본은 내려받지 않으며, 감사 로그에는 검출 내용 없이 "검사 사실(파일키·시각)"만 기록한다.

#### 15. 향후 확장 로드맵 (갱신)

XD는 Adobe의 사양 축소 추세를 고려해 후순위로 유지하고, Figma 옵트인 검사는 디자이너 수요에 따라 점진 강화한다. Sketch 등 추가 디자인 포맷 지원은 수요 검토 후 결정한다.

## 2. PSD/XD/Figma 검사 단위 테스트

검증의 핵심은 텍스트 레이어 추출 정확성, XD JSON 파싱 안정성, 그리고 Figma 옵트인 가드가 동의·토큰 없이는 절대 동작하지 않음을 확인하는 것입니다.

### 2-1. 디자인 추출기 테스트 — `tests/test_design_extractors.py`

```python
# tests/test_design_extractors.py
"""PSD/XD 디자인 파일 추출기 테스트"""

import json
import zipfile
import pytest
from pathlib import Path

from soliguard.design_extractors import (
    extract_design_text, _extract_xd, _walk_xd_json, SUPPORTED_DESIGN,
)
from soliguard.extractors import ExtractionError


# ── XD: 표준 라이브러리만으로 테스트 가능(임시 zip 생성) ──
class TestXD:
    def _make_xd(self, tmp_path: Path, json_obj: dict) -> Path:
        """가짜 .xd(ZIP) 파일 생성"""
        p = tmp_path / "design.xd"
        with zipfile.ZipFile(p, "w") as z:
            z.writestr("artwork/artboard1.json", json.dumps(json_obj, ensure_ascii=False))
        return p

    def test_extract_text_node(self, tmp_path):
        doc = {"children": [
            {"type": "text", "text": {"rawText": "고객 홍길동 900101-1234567"}},
            {"type": "shape"},
        ]}
        p = self._make_xd(tmp_path, doc)
        text = _extract_xd(p)
        assert "900101-1234567" in text

    def test_nested_text_nodes(self, tmp_path):
        doc = {"children": [
            {"children": [
                {"type": "text", "text": {"rawText": "010-1234-5678"}},
            ]},
        ]}
        p = self._make_xd(tmp_path, doc)
        assert "010-1234-5678" in _extract_xd(p)

    def test_invalid_zip_raises(self, tmp_path):
        p = tmp_path / "broken.xd"
        p.write_bytes(b"not a zip file")
        with pytest.raises(ExtractionError):
            _extract_xd(p)

    def test_ignores_non_json_entries(self, tmp_path):
        p = tmp_path / "mixed.xd"
        with zipfile.ZipFile(p, "w") as z:
            z.writestr("resources/image.png", b"\x89PNG\r\n")  # 바이너리
            z.writestr("artwork/a.json", json.dumps(
                {"type": "text", "text": {"rawText": "test@example.com"}}))
        assert "test@example.com" in _extract_xd(p)


# ── XD JSON 워커 단위 테스트 ──
class TestXDWalker:
    def test_collects_rawtext(self):
        acc = []
        _walk_xd_json({"type": "text", "text": {"rawText": "민감정보"}}, acc)
        assert "민감정보" in acc

    def test_collects_direct_text_key(self):
        acc = []
        _walk_xd_json({"rawText": "직접키"}, acc)
        assert "직접키" in acc

    def test_handles_list(self):
        acc = []
        _walk_xd_json([{"text": {"value": "리스트항목"}}], acc)
        assert "리스트항목" in acc

    def test_empty_safe(self):
        acc = []
        _walk_xd_json({}, acc)
        _walk_xd_json(None, acc)
        assert acc == []


# ── PSD: psd-tools 설치 시에만 (실제 PSD 생성) ──
class TestPSD:
    def test_psd_text_layer(self, tmp_path):
        psd_tools = pytest.importorskip("psd_tools")
        # psd-tools로 텍스트 레이어가 있는 PSD를 생성하기는 복잡하므로,
        # 여기서는 라이브러리 미설치 시 적절한 예외를 확인하는 수준으로 검증.
        # 실제 PSD 픽스처는 tests/fixtures/sample_with_text.psd 로 별도 제공 권장.
        fixture = Path(__file__).parent / "fixtures" / "sample_with_text.psd"
        if not fixture.exists():
            pytest.skip("PSD 픽스처 없음 - CI에 sample PSD 추가 필요")
        text = extract_design_text(fixture, ocr_enabled=False)
        assert isinstance(text, str)

    def test_unsupported_extension(self, tmp_path):
        p = tmp_path / "x.sketch"
        p.write_bytes(b"\x00")
        with pytest.raises(ExtractionError):
            extract_design_text(p)


def test_supported_design_set():
    assert ".psd" in SUPPORTED_DESIGN
    assert ".xd" in SUPPORTED_DESIGN
```

### 2-2. Figma 옵트인 모듈 테스트 — `tests/test_figma_scan.py`

Figma 테스트의 핵심은 동의·토큰 가드와, 실제 API를 호출하지 않도록 응답을 모킹(mock)하는 것입니다.

```python
# tests/test_figma_scan.py
"""Figma 옵트인 검사 모듈 테스트 - 가드 + 텍스트노드 수집 + 검출"""

import pytest
from unittest.mock import patch, MagicMock

from soliguard.figma_scan import (
    scan_figma_file, parse_file_key, _collect_text_nodes,
    FigmaConsentError, FigmaApiError,
)


# ── 옵트인 가드: 가장 중요 (동의/토큰 없으면 절대 동작 안 함) ──
class TestConsentGuard:
    def test_no_consent_blocked(self):
        with pytest.raises(FigmaConsentError):
            scan_figma_file("KEY", "token", user_consented=False)

    def test_no_token_blocked(self):
        with pytest.raises(FigmaConsentError):
            scan_figma_file("KEY", "", user_consented=True)

    def test_guard_runs_before_network(self):
        # 가드 실패 시 네트워크 호출이 일어나지 않아야 함
        with patch("requests.get") as mock_get:
            with pytest.raises(FigmaConsentError):
                scan_figma_file("KEY", "", user_consented=False)
            mock_get.assert_not_called()


# ── URL 파싱 ──
class TestParseFileKey:
    def test_file_url(self):
        assert parse_file_key("https://www.figma.com/file/AbC123/제목") == "AbC123"

    def test_design_url(self):
        assert parse_file_key("https://figma.com/design/XyZ789/x") == "XyZ789"

    def test_invalid_url(self):
        with pytest.raises(ValueError):
            parse_file_key("https://example.com/foo")


# ── 텍스트 노드 수집: TEXT 타입의 characters만 ──
class TestCollectTextNodes:
    def test_collects_text_type(self):
        doc = {"type": "DOCUMENT", "children": [
            {"type": "TEXT", "characters": "주민번호 900101-1234567"},
            {"type": "RECTANGLE"},  # 텍스트 아님 → 무시
        ]}
        acc = []
        _collect_text_nodes(doc, acc)
        assert acc == ["주민번호 900101-1234567"]

    def test_nested(self):
        doc = {"children": [
            {"type": "FRAME", "children": [
                {"type": "TEXT", "characters": "010-1234-5678"},
            ]},
        ]}
        acc = []
        _collect_text_nodes(doc, acc)
        assert "010-1234-5678" in acc

    def test_ignores_empty(self):
        acc = []
        _collect_text_nodes({"type": "TEXT", "characters": "   "}, acc)
        assert acc == []


# ── API 응답 모킹: 실제 검출까지 ──
class TestScanWithMock:
    def _mock_response(self, status=200, doc=None, name="테스트파일"):
        resp = MagicMock()
        resp.status_code = status
        resp.json.return_value = {"name": name, "document": doc or {}}
        return resp

    def test_successful_scan_detects(self):
        doc = {"type": "DOCUMENT", "children": [
            {"type": "TEXT", "characters": "카드 4111-1111-1111-1111"},
        ]}
        with patch("requests.get", return_value=self._mock_response(doc=doc)):
            with patch("soliguard.figma_scan._audit_figma"):  # 로그 격리
                result = scan_figma_file("KEY", "valid-token", user_consented=True)
        kinds = [f.kind for f in result.findings]
        assert "신용카드번호" in kinds
        assert result.text_node_count == 1

    def test_403_raises_api_error(self):
        with patch("requests.get", return_value=self._mock_response(status=403)):
            with pytest.raises(FigmaApiError):
                scan_figma_file("KEY", "bad-token", user_consented=True)

    def test_404_raises_api_error(self):
        with patch("requests.get", return_value=self._mock_response(status=404)):
            with pytest.raises(FigmaApiError):
                scan_figma_file("BADKEY", "token", user_consented=True)

    def test_returns_masked_only(self):
        # 결과의 findings는 마스킹된 값을 보유(원본 텍스트 미반환 확인)
        doc = {"type": "DOCUMENT", "children": [
            {"type": "TEXT", "characters": "4111-1111-1111-1111"},
        ]}
        with patch("requests.get", return_value=self._mock_response(doc=doc)):
            with patch("soliguard.figma_scan._audit_figma"):
                result = scan_figma_file("KEY", "token", user_consented=True)
        # FigmaScanResult에는 raw 원본 텍스트 필드가 없어야 함
        assert not hasattr(result, "raw_text")
        for f in result.findings:
            assert "*" in f.masked  # 마스킹 적용됨
```

테스트의 핵심 가치는 세 가지입니다. 첫째, `test_guard_runs_before_network`는 동의·토큰이 없으면 네트워크 호출조차 일어나지 않음을 검증해 옵트인 원칙이 코드 레벨에서 강제됨을 보장합니다. 둘째, API를 모킹해 실제 Figma 서버 없이도 검출 파이프라인 전체를 테스트합니다. 셋째, `test_returns_masked_only`로 결과에 원본 텍스트가 남지 않고 마스킹된 값만 반환됨을 확인해 비저장 원칙을 보증합니다. XD 테스트는 표준 라이브러리로 임시 파일을 만들어 외부 픽스처 없이 동작하고, PSD는 실제 픽스처가 필요하므로 CI에 샘플 PSD를 두고 없으면 스킵하도록 했습니다.

두 작업이 모두 완료됐습니다. 직무 프로파일은 디자이너가 디자인 파일을 우선·OCR 포함으로 점검하도록 구성됐고, 사양서 항목은 PSD·XD 지원과 Figma 옵트인 정책을 반영했으며, 단위 테스트는 추출 정확성과 옵트인 가드를 검증합니다.
