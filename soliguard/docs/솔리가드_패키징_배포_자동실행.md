## 3. 설정 관리 모듈 (`config.py`)

스케줄과 사용자 설정을 OS 표준 경로에 저장합니다.

```python
# src/soliguard/config.py
"""사용자 설정 로드/저장 - OS 표준 경로에 JSON 보관"""

import json
from pathlib import Path
from dataclasses import dataclass, asdict, field
from platformdirs import user_config_dir, user_data_dir

CONFIG_DIR = Path(user_config_dir("SoliGuard", "Solideo"))
DATA_DIR = Path(user_data_dir("SoliGuard", "Solideo"))
CONFIG_FILE = CONFIG_DIR / "config.json"


@dataclass
class ScheduleConfig:
    enabled: bool = False
    frequency: str = "weekly"          # "daily" | "weekly" | "monthly"
    day_of_week: str = "mon"           # weekly용 (mon~sun)
    day_of_month: int = 1              # monthly용
    hour: int = 9
    minute: int = 0


@dataclass
class AppConfig:
    profile: str = "개발자"
    target_folders: list[str] = field(default_factory=list)
    ocr_mode: str = "local"            # "local" | "cloud"(동의 시)
    exclude_folders: list[str] = field(default_factory=list)
    schedule: ScheduleConfig = field(default_factory=ScheduleConfig)
    auto_action: str = "report_only"   # "report_only" | "quarantine"
    # 자동 스캔 시 발견만 할지(report_only), 위험 파일 자동 격리까지 할지

    @classmethod
    def load(cls) -> "AppConfig":
        if CONFIG_FILE.exists():
            data = json.loads(CONFIG_FILE.read_text(encoding="utf-8"))
            sched = ScheduleConfig(**data.pop("schedule", {}))
            return cls(schedule=sched, **data)
        return cls()

    def save(self):
        CONFIG_DIR.mkdir(parents=True, exist_ok=True)
        data = asdict(self)
        CONFIG_FILE.write_text(
            json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")
```

`auto_action`은 중요한 안전 설계입니다. 자동 스캔에서 곧바로 파일을 삭제하면 사고 위험이 크므로, 기본값은 "발견만 하고 리포트 생성"이며, 사용자가 명시적으로 설정해야 자동 격리까지 갑니다. 자동 완전삭제는 무인 실행에서 절대 하지 않습니다.

## 4. 정기 자동 스캔 스케줄러 (`scheduler.py`)

APScheduler로 백그라운드 에이전트를 구성합니다. 설정된 주기에 스캔을 돌리고 결과를 리포트로 남기며, 사용자에게 알림을 띄웁니다.

```python
# src/soliguard/scheduler.py
"""정기 자동 스캔 에이전트 - 백그라운드에서 주기 실행"""

import sys
import logging
from pathlib import Path
from datetime import datetime

from apscheduler.schedulers.blocking import BlockingScheduler
from apscheduler.triggers.cron import CronTrigger

from soliguard.config import AppConfig, DATA_DIR
from soliguard.engine import run_scan
from soliguard.report import generate_pdf_report

logging.basicConfig(
    filename=str(DATA_DIR / "agent.log"),
    level=logging.INFO,
    format="%(asctime)s %(levelname)s %(message)s",
)
log = logging.getLogger("soliguard.agent")


def _build_trigger(sched) -> CronTrigger:
    """설정을 APScheduler cron 트리거로 변환"""
    if sched.frequency == "daily":
        return CronTrigger(hour=sched.hour, minute=sched.minute)
    if sched.frequency == "weekly":
        return CronTrigger(day_of_week=sched.day_of_week,
                           hour=sched.hour, minute=sched.minute)
    if sched.frequency == "monthly":
        return CronTrigger(day=sched.day_of_month,
                           hour=sched.hour, minute=sched.minute)
    raise ValueError(f"알 수 없는 주기: {sched.frequency}")


def run_scheduled_scan():
    """예약된 시점에 실행되는 스캔 작업"""
    cfg = AppConfig.load()
    folders = [Path(f) for f in cfg.target_folders]
    if not folders:
        log.warning("스캔 대상 폴더가 설정되지 않음 - 건너뜀")
        return

    log.info("정기 스캔 시작: %s", folders)
    summary = run_scan(folders, profile=cfg.profile,
                       ocr_enabled=(cfg.ocr_mode != "off"),
                       excludes=set(cfg.exclude_folders))
    log.info("스캔 완료: 발견 %d건, 등급 %s",
             summary.total_findings, summary.risk_grade)

    # 리포트 자동 저장
    DATA_DIR.mkdir(parents=True, exist_ok=True)
    report_path = DATA_DIR / f"report_{datetime.now():%Y%m%d_%H%M}.pdf"
    generate_pdf_report(summary, profile=cfg.profile, output_path=report_path)

    # 자동 격리(설정된 경우에만, 완전삭제는 절대 자동 실행 안 함)
    if cfg.auto_action == "quarantine" and summary.risk_grade != "safe":
        from soliguard.actions import quarantine_file
        for r in summary.file_results:
            if any(f.severity.value == "높음" for f in r.findings):
                quarantine_file(Path(r.path))
                log.info("자동 격리: %s", r.path)

    _notify(summary, report_path)


def _notify(summary, report_path):
    """사용자에게 결과 알림 (윈도우 토스트)"""
    try:
        from plyer import notification  # 선택 의존성
        notification.notify(
            title="SoliGuard 정기 점검 완료",
            message=f"위험 {summary.total_findings}건 발견 (등급: {summary.risk_grade})",
            timeout=10,
        )
    except Exception as e:
        log.info("알림 표시 실패(무시): %s", e)


def run_agent():
    """백그라운드 에이전트 진입점 - 설정 주기대로 스캔 예약"""
    cfg = AppConfig.load()
    if not cfg.schedule.enabled:
        log.info("스케줄 비활성 상태 - 에이전트 종료")
        return

    scheduler = BlockingScheduler()
    trigger = _build_trigger(cfg.schedule)
    scheduler.add_job(run_scheduled_scan, trigger, id="periodic_scan",
                      misfire_grace_time=3600)  # PC가 꺼져 놓친 작업도 1시간 내 보정
    log.info("에이전트 시작 - 주기: %s", cfg.schedule.frequency)
    try:
        scheduler.start()
    except (KeyboardInterrupt, SystemExit):
        log.info("에이전트 종료")


if __name__ == "__main__":
    run_agent()
```

스케줄러의 핵심 안전장치는 자동 실행에서 완전삭제를 절대 하지 않는다는 점, 기본값이 "리포트만 생성"이라는 점, 그리고 PC가 꺼져 있어 놓친 작업을 `misfire_grace_time`으로 보정하는 점입니다.

```python
# build_exe.spec
# 빌드 명령: pyinstaller build_exe.spec
from PyInstaller.utils.hooks import collect_data_files, collect_submodules

block_cipher = None

# 공통 hidden imports (동적 import 모듈)
hidden = (collect_submodules("apscheduler")
          + collect_submodules("pdfplumber")
          + ["pytesseract", "openpyxl", "docx", "olefile", "reportlab"])

datas = collect_data_files("reportlab")
datas += [("assets/soliguard.ico", "assets")]
# Tesseract 데이터(kor)를 번들하려면 아래처럼 추가:
# datas += [("vendor/tesseract", "tesseract")]

# 1) GUI 앱
gui_a = Analysis(["src/soliguard/gui.py"], pathex=["src"],
                 datas=datas, hiddenimports=hidden, cipher=block_cipher)
gui_pyz = PYZ(gui_a.pure)
gui_exe = EXE(gui_pyz, gui_a.scripts, [], exclude_binaries=True,
              name="SoliGuard", console=False, icon="assets/soliguard.ico")

# 2) 백그라운드 에이전트(스케줄러)
agent_a = Analysis(["src/soliguard/scheduler.py"], pathex=["src"],
                   datas=datas, hiddenimports=hidden, cipher=block_cipher)
agent_pyz = PYZ(agent_a.pure)
agent_exe = EXE(agent_pyz, agent_a.scripts, [], exclude_binaries=True,
                name="SoliGuardAgent", console=False, icon="assets/soliguard.ico")

coll = COLLECT(gui_exe, gui_a.binaries, gui_a.datas,
               agent_exe, agent_a.binaries, agent_a.datas,
               name="SoliGuard")
```

## 6. 자동 실행 등록 — 정해진 주기로 작동하게 하기

exe가 정해진 주기에 자동으로 도는 방식은 두 가지입니다. 권장하는 방식은 **Windows 작업 스케줄러(Task Scheduler)**에 등록하는 것입니다. APScheduler를 상주시키는 것보다, OS 스케줄러가 정해진 시각에 에이전트를 깨우는 편이 PC 재부팅·절전에도 안정적이기 때문입니다.

설치 시 또는 설정에서 스케줄을 켤 때, 다음처럼 작업 스케줄러에 등록하는 헬퍼를 둡니다.

```python
# src/soliguard/scheduler.py (이어서) - Windows 작업 등록 헬퍼
import subprocess
import sys
from pathlib import Path


def register_windows_task(cfg) -> bool:
    """Windows 작업 스케줄러에 정기 스캔 작업 등록.
    OS가 정해진 시각에 SoliGuardAgent.exe를 1회 실행하도록 함."""
    sched = cfg.schedule
    agent_exe = Path(sys.executable).parent / "SoliGuardAgent.exe"
    # 개발 환경에서는 'python -m soliguard.scheduler'로 대체 가능

    # schtasks 주기 매핑
    if sched.frequency == "daily":
        sc = ["/SC", "DAILY"]
    elif sched.frequency == "weekly":
        sc = ["/SC", "WEEKLY", "/D", sched.day_of_week.upper()]
    elif sched.frequency == "monthly":
        sc = ["/SC", "MONTHLY", "/D", str(sched.day_of_month)]
    else:
        return False

    cmd = [
        "schtasks", "/Create", "/TN", "SoliGuard_PeriodicScan",
        "/TR", f'"{agent_exe}" --once',     # 한 번 스캔하고 종료
        *sc,
        "/ST", f"{sched.hour:02d}:{sched.minute:02d}",
        "/F",   # 기존 작업 덮어쓰기
    ]
    try:
        subprocess.run(cmd, check=True, capture_output=True)
        return True
    except subprocess.CalledProcessError:
        return False


def unregister_windows_task() -> bool:
    try:
        subprocess.run(["schtasks", "/Delete", "/TN", "SoliGuard_PeriodicScan", "/F"],
                       check=True, capture_output=True)
        return True
    except subprocess.CalledProcessError:
        return False
```

이 방식에서는 에이전트가 `--once` 인자를 받으면 스케줄러로 상주하지 않고 한 번만 스캔하고 종료하도록 `run_agent`를 분기합니다. 즉 OS 작업 스케줄러가 "주기"를 담당하고, exe는 "1회 스캔"만 책임지는 깔끔한 구조입니다.

## 7. 전체 동작 흐름 정리

설치부터 자동 실행까지의 흐름은 이렇습니다. 사용자가 `SoliGuard_Setup.exe`를 실행해 설치하면서 "주 1회 자동 점검"을 체크하면, 인스톨러가 GUI/에이전트 exe를 설치하고 Windows 작업 스케줄러에 매주 월요일 9시 작업을 등록합니다. 이후 매주 그 시각에 OS가 `SoliGuardAgent.exe --once`를 깨워 설정된 폴더를 스캔하고, 결과를 PDF 리포트로 저장한 뒤 토스트 알림을 띄웁니다. 완전삭제는 자동으로 절대 일어나지 않고, 기본은 발견·리포트만 하며 사용자가 원할 때만 자동 격리가 동작합니다. 사용자는 GUI에서 직무·대상 폴더·주기·자동 조치 수준을 언제든 바꿀 수 있고, 변경 시 `register_windows_task()`가 작업을 갱신합니다.

테스트는 스케줄러의 트리거 변환을 검증하는 케이스를 더합니다.

```python
# tests/test_scheduler.py
from soliguard.config import ScheduleConfig
from soliguard.scheduler import _build_trigger

def test_weekly_trigger():
    sched = ScheduleConfig(frequency="weekly", day_of_week="mon", hour=9, minute=0)
    trigger = _build_trigger(sched)
    assert "mon" in str(trigger)

def test_daily_trigger():
    trigger = _build_trigger(ScheduleConfig(frequency="daily", hour=22))
    assert trigger is not None
```

이제 솔리가드는 설치형 패키지로 빌드되어 정해진 주기에 자동으로 PC를 점검하고 리포트를 남기는 완전한 형태를 갖췄습니다. 코드 모듈, 빌드 스펙, 설치 스크립트, 자동 실행 등록까지 실제 제작에 바로 착수할 수 있는 수준입니다.

> **실무 참고**: 이 도구는 사용자 본인/사내 PC를 정당하게 점검하는 목적이며, 작업 스케줄러 등록과 파일 격리·삭제는 관리자 권한과 데이터 접근을 수반하므로 사내 배포 시 보안팀 검토와 사용자 고지(어떤 폴더를 점검하고 무엇을 어떻게 처리하는지)를 함께 진행하는 것이 좋습니다. 또한 OCR을 쓰려면 Tesseract와 poppler를 인스톨러에 번들하거나 설치 단계에서 함께 배포해야 합니다.
