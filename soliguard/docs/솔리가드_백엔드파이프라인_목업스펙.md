# 솔리가드(SoliGuard) 백엔드 파이프라인 & 목업 스펙

파일 포맷별 텍스트 추출기, 조치 계층(마스킹·격리·완전삭제), 고해상도 목업 스펙을 정리한 문서입니다.

---

## 1. 파일 포맷별 텍스트 추출기 (`extractors.py`)

검출 엔진이 통일된 인터페이스 `extract_text(path) -> str`로 모든 포맷을 텍스트로 받을 수 있게 하는 모듈입니다. 포맷별 라이브러리를 쓰되, 한 파일의 파싱 실패가 전체 스캔을 멈추지 않도록 예외를 격리합니다.

### 1-1. 진입점 & 포맷 분기

```python
# soliguard/extractors.py
"""파일 포맷별 텍스트 추출 - 검출 엔진에 통일된 텍스트를 공급"""

from pathlib import Path

# 사용 라이브러리(설치 필요):
#   openpyxl, xlrd(구형 xls), python-docx, pdfplumber, olefile/pyhwp 또는 hwp5,
#   Pillow + pytesseract(OCR), chardet


class ExtractionError(Exception):
    """파싱 불가(암호/손상/미지원) 시 발생 → 상위에서 '검사불가' 처리"""


SUPPORTED_TEXT = {".txt", ".csv", ".log", ".json", ".xml", ".md",
                  ".py", ".java", ".js", ".sql", ".yml", ".yaml", ".properties", ".env"}
SUPPORTED_OFFICE = {".xlsx", ".xls", ".docx"}
SUPPORTED_HWP = {".hwp", ".hwpx"}
SUPPORTED_PDF = {".pdf"}
SUPPORTED_IMAGE = {".jpg", ".jpeg", ".png", ".bmp", ".tiff"}


def extract_text(path: Path, ocr_enabled: bool = True) -> str:
    """확장자에 맞는 추출기로 분기. 실패 시 ExtractionError."""
    path = Path(path)
    ext = path.suffix.lower()

    try:
        if ext in SUPPORTED_TEXT:
            return _extract_plain(path)
        if ext == ".xlsx":
            return _extract_xlsx(path)
        if ext == ".xls":
            return _extract_xls(path)
        if ext == ".docx":
            return _extract_docx(path)
        if ext in SUPPORTED_HWP:
            return _extract_hwp(path)
        if ext == ".pdf":
            return _extract_pdf(path, ocr_enabled)
        if ext in SUPPORTED_IMAGE:
            return _extract_image(path) if ocr_enabled else ""
        raise ExtractionError(f"미지원 형식: {ext}")
    except ExtractionError:
        raise
    except Exception as e:
        raise ExtractionError(f"{path.name} 파싱 실패: {e}") from e
```

### 1-2. 포맷별 추출 함수

```python
# soliguard/extractors.py (이어서)

def _extract_plain(path: Path) -> str:
    """텍스트/소스/로그 - 인코딩 자동 감지(한글 cp949 대응)"""
    import chardet
    raw = path.read_bytes()
    if not raw:
        return ""
    enc = chardet.detect(raw[:100_000]).get("encoding") or "utf-8"
    try:
        return raw.decode(enc, errors="replace")
    except LookupError:
        return raw.decode("utf-8", errors="replace")


def _extract_xlsx(path: Path) -> str:
    """엑셀(xlsx) - 모든 시트의 셀 값을 텍스트로. 대용량은 read_only."""
    import openpyxl
    wb = openpyxl.load_workbook(path, read_only=True, data_only=True)
    parts = []
    for ws in wb.worksheets:
        for row in ws.iter_rows(values_only=True):
            cells = [str(c) for c in row if c is not None]
            if cells:
                parts.append(" ".join(cells))
    wb.close()
    return "\n".join(parts)


def _extract_xls(path: Path) -> str:
    """구형 엑셀(xls)"""
    import xlrd
    book = xlrd.open_workbook(path)
    parts = []
    for sheet in book.sheets():
        for r in range(sheet.nrows):
            cells = [str(c) for c in sheet.row_values(r) if c != ""]
            if cells:
                parts.append(" ".join(cells))
    return "\n".join(parts)


def _extract_docx(path: Path) -> str:
    """워드(docx) - 본문 단락 + 표 셀"""
    import docx
    doc = docx.Document(path)
    parts = [p.text for p in doc.paragraphs if p.text.strip()]
    for table in doc.tables:
        for row in table.rows:
            cells = [cell.text for cell in row.cells if cell.text.strip()]
            if cells:
                parts.append(" ".join(cells))
    return "\n".join(parts)


def _extract_hwp(path: Path) -> str:
    """한글(hwp/hwpx) - 국내 SI 환경 필수.
    hwpx는 zip+xml, 구형 hwp는 OLE 복합문서(BodyText 스트림)."""
    if path.suffix.lower() == ".hwpx":
        return _extract_hwpx(path)
    return _extract_hwp_ole(path)


def _extract_hwpx(path: Path) -> str:
    """hwpx: zip 내부 Contents/section*.xml에서 텍스트 추출"""
    import zipfile
    import re
    parts = []
    with zipfile.ZipFile(path) as z:
        for name in z.namelist():
            if name.startswith("Contents/section") and name.endswith(".xml"):
                xml = z.read(name).decode("utf-8", errors="replace")
                # <hp:t> 태그 안의 텍스트만 추출
                parts += re.findall(r"<hp:t>(.*?)</hp:t>", xml, re.DOTALL)
    return "\n".join(re.sub(r"<[^>]+>", "", p) for p in parts)


def _extract_hwp_ole(path: Path) -> str:
    """구형 hwp: OLE BodyText 스트림(zlib 압축) 해제 후 텍스트 추출.
    암호 설정 문서는 ExtractionError."""
    import olefile
    import zlib
    import struct

    if not olefile.isOleFile(path):
        raise ExtractionError("유효한 HWP(OLE) 파일이 아님")
    ole = olefile.OleFileIO(path)
    try:
        # FileHeader로 압축/암호 여부 확인
        header = ole.openstream("FileHeader").read()
        is_compressed = bool(header[36] & 0x01)
        is_encrypted = bool(header[36] & 0x02)
        if is_encrypted:
            raise ExtractionError("암호로 보호된 HWP 문서")

        texts = []
        for entry in ole.listdir():
            if entry[0] == "BodyText":
                data = ole.openstream(entry).read()
                if is_compressed:
                    data = zlib.decompress(data, -15)
                texts.append(_decode_hwp_records(data))
        return "\n".join(texts)
    finally:
        ole.close()


def _decode_hwp_records(data: bytes) -> str:
    """HWP 레코드 구조에서 텍스트(UTF-16LE) 추출 (간략 구현).
    실제로는 레코드 태그(HWPTAG_PARA_TEXT=67) 파싱 권장."""
    import struct
    out, i, n = [], 0, len(data)
    while i + 4 <= n:
        header = struct.unpack("<I", data[i:i + 4])[0]
        tag = header & 0x3FF
        size = (header >> 20) & 0xFFF
        i += 4
        if tag == 67:  # HWPTAG_PARA_TEXT
            chunk = data[i:i + size]
            out.append(chunk.decode("utf-16-le", errors="replace"))
        i += size
    return "".join(out)


def _extract_pdf(path: Path, ocr_enabled: bool) -> str:
    """PDF - 텍스트 레이어 우선, 없으면(스캔 PDF) OCR 폴백"""
    import pdfplumber
    parts = []
    with pdfplumber.open(path) as pdf:
        for page in pdf.pages:
            t = page.extract_text() or ""
            parts.append(t)
    text = "\n".join(parts).strip()
    if not text and ocr_enabled:
        return _ocr_pdf(path)
    return text


def _extract_image(path: Path) -> str:
    """이미지 OCR - 신분증·계약서 스캔본 (로컬 Tesseract 기본)"""
    from PIL import Image
    import pytesseract
    img = Image.open(path)
    # 한국어+영어, 회전 보정은 전처리 단계에서 추가 가능
    return pytesseract.image_to_string(img, lang="kor+eng")


def _ocr_pdf(path: Path) -> str:
    """스캔 PDF를 이미지로 변환 후 OCR"""
    from pdf2image import convert_from_path
    import pytesseract
    parts = [pytesseract.image_to_string(img, lang="kor+eng")
             for img in convert_from_path(path, dpi=200)]
    return "\n".join(parts)
```

이 모듈의 핵심은 통일된 `extract_text()` 진입점, 한글 cp949 인코딩 자동 감지, hwp/hwpx 분기 처리(국내 차별점), 그리고 텍스트 레이어가 없는 스캔 PDF·이미지의 OCR 폴백입니다. 암호 걸린 hwp는 `ExtractionError`를 던져 상위에서 "검사불가"로 분리 기록됩니다.

## 2. 조치 계층 코드 (마스킹·격리·완전삭제)

검출 후 사용자가 선택하는 세 가지 조치를 구현합니다. 안전성을 위해 격리는 암호화하고, 완전삭제는 덮어쓰기 후 삭제하며, 모든 조치는 감사 로그에 남깁니다.

### 2-1. 감사 로그 & 기본 구조

```python
# soliguard/actions.py
"""조치 계층: 마스킹 / 암호화 격리 / 완전삭제(wiping) + 감사 로그"""

import os
import json
import shutil
import secrets
from pathlib import Path
from datetime import datetime
from dataclasses import dataclass

# 암호화: cryptography 라이브러리 (AES-256-GCM)
from cryptography.hazmat.primitives.ciphers.aead import AESGCM


QUARANTINE_DIR = Path.home() / ".soliguard" / "quarantine"
AUDIT_LOG = Path.home() / ".soliguard" / "audit.log"


@dataclass
class ActionResult:
    action: str          # "mask" | "quarantine" | "delete"
    path: str
    status: str          # "success" | "failed"
    detail: str = ""


# ---------------------------------------------------------------------------
# 감사 로그 (컴플라이언스 증빙)
# ---------------------------------------------------------------------------

def write_audit(action: str, path: str, result: str, extra: dict | None = None):
    """모든 조치를 추가 전용(append-only) 로그로 기록"""
    AUDIT_LOG.parent.mkdir(parents=True, exist_ok=True)
    record = {
        "ts": datetime.now().isoformat(timespec="seconds"),
        "action": action,
        "path": path,
        "result": result,
        "user": os.getenv("USERNAME") or os.getenv("USER", "unknown"),
        **(extra or {}),
    }
    with open(AUDIT_LOG, "a", encoding="utf-8") as f:
        f.write(json.dumps(record, ensure_ascii=False) + "\n")
```

### 2-2. 마스킹 — 검출 부분만 가린 사본 생성

```python
# ---------------------------------------------------------------------------
# 1) 마스킹: 검출 부분만 가린 사본 생성 (원본 보존 옵션)
# ---------------------------------------------------------------------------

def mask_in_text_file(path: Path, findings: list, in_place: bool = False) -> ActionResult:
    """텍스트 파일 내 검출 문자열을 마스킹 값으로 치환.
    findings: Finding 리스트 (raw, masked 보유)"""
    try:
        content = path.read_text(encoding="utf-8", errors="replace")
        # 긴 문자열부터 치환해 부분겹침 방지
        for f in sorted(findings, key=lambda x: -len(x.raw)):
            content = content.replace(f.raw, f.masked)

        target = path if in_place else path.with_name(path.stem + "_masked" + path.suffix)
        target.write_text(content, encoding="utf-8")
        write_audit("mask", str(path), "success",
                    {"output": str(target), "count": len(findings)})
        return ActionResult("mask", str(path), "success", str(target))
    except Exception as e:
        write_audit("mask", str(path), "failed", {"error": str(e)})
        return ActionResult("mask", str(path), "failed", str(e))
```

### 2-3. 격리 — AES-256 암호화 이동 & 복원

```python
# ---------------------------------------------------------------------------
# 2) 격리: 파일을 AES-256으로 암호화해 격리 폴더로 이동 (복원 가능)
# ---------------------------------------------------------------------------

def quarantine_file(path: Path) -> ActionResult:
    """원본을 암호화 후 격리함으로 이동. 메타데이터로 복원 정보 보관."""
    try:
        QUARANTINE_DIR.mkdir(parents=True, exist_ok=True)
        key = AESGCM.generate_key(bit_length=256)
        nonce = secrets.token_bytes(12)
        aesgcm = AESGCM(key)

        data = path.read_bytes()
        encrypted = aesgcm.encrypt(nonce, data, None)

        qid = secrets.token_hex(8)
        (QUARANTINE_DIR / f"{qid}.enc").write_bytes(nonce + encrypted)
        # 복원용 메타 (키는 OS 키체인/별도 보관 권장 - 여기선 단순화)
        meta = {
            "id": qid,
            "original_path": str(path),
            "quarantined_at": datetime.now().isoformat(timespec="seconds"),
            "size": len(data),
            "key": key.hex(),     # 실제 제품에서는 OS 보안 저장소(DPAPI 등)에 분리 저장
        }
        (QUARANTINE_DIR / f"{qid}.meta.json").write_text(
            json.dumps(meta, ensure_ascii=False), encoding="utf-8")

        # 원본은 안전 삭제 후 제거
        _secure_overwrite(path)
        path.unlink()

        write_audit("quarantine", str(path), "success", {"qid": qid})
        return ActionResult("quarantine", str(path), "success", qid)
    except Exception as e:
        write_audit("quarantine", str(path), "failed", {"error": str(e)})
        return ActionResult("quarantine", str(path), "failed", str(e))


def restore_file(qid: str) -> ActionResult:
    """격리 파일 복원"""
    try:
        meta = json.loads((QUARANTINE_DIR / f"{qid}.meta.json").read_text(encoding="utf-8"))
        blob = (QUARANTINE_DIR / f"{qid}.enc").read_bytes()
        nonce, encrypted = blob[:12], blob[12:]
        aesgcm = AESGCM(bytes.fromhex(meta["key"]))
        data = aesgcm.decrypt(nonce, encrypted, None)

        target = Path(meta["original_path"])
        target.parent.mkdir(parents=True, exist_ok=True)
        target.write_bytes(data)

        (QUARANTINE_DIR / f"{qid}.enc").unlink()
        (QUARANTINE_DIR / f"{qid}.meta.json").unlink()
        write_audit("restore", meta["original_path"], "success", {"qid": qid})
        return ActionResult("restore", meta["original_path"], "success")
    except Exception as e:
        write_audit("restore", qid, "failed", {"error": str(e)})
        return ActionResult("restore", qid, "failed", str(e))
```

### 2-4. 완전삭제 — 덮어쓰기(wiping) 후 삭제

```python
# ---------------------------------------------------------------------------
# 3) 완전삭제: 덮어쓰기(wiping) 후 삭제 - 복구 불가
# ---------------------------------------------------------------------------

def _secure_overwrite(path: Path, passes: int = 3):
    """파일 내용을 난수로 여러 번 덮어써 복구를 어렵게 함.
    (SSD는 wear-leveling으로 한계가 있어, 격리암호화를 1차 권장)"""
    length = path.stat().st_size
    if length == 0:
        return
    with open(path, "r+b", buffering=0) as f:
        for _ in range(passes):
            f.seek(0)
            f.write(secrets.token_bytes(length))
            f.flush()
            os.fsync(f.fileno())


def secure_delete(path: Path, confirmed: bool = False) -> ActionResult:
    """완전삭제. confirmed=True(UI 확인 게이트 통과) 일 때만 실행."""
    if not confirmed:
        return ActionResult("delete", str(path), "failed", "확인 미통과")
    try:
        _secure_overwrite(path)
        path.unlink()
        write_audit("delete", str(path), "success")
        return ActionResult("delete", str(path), "success")
    except Exception as e:
        write_audit("delete", str(path), "failed", {"error": str(e)})
        return ActionResult("delete", str(path), "failed", str(e))
```

조치 계층의 핵심 설계 의도는 세 가지입니다. 격리는 AES-256-GCM으로 암호화해 보관하고 복원 가능하게 하되, 키는 실제 제품에서 OS 보안 저장소(Windows DPAPI 등)에 분리 저장해야 합니다(코드에는 단순화해 표시). 완전삭제는 `confirmed=True`일 때만 실행되어 UI의 확인 게이트와 연결되며, SSD에서는 물리적 복구 차단에 한계가 있으므로 격리 암호화를 1차 권장합니다. 모든 조치는 append-only 감사 로그에 남아 컴플라이언스 증빙이 됩니다.

## 3. 고해상도 목업 스펙

디자인 가이드를 실제 화면 픽셀·간격·컴포넌트 배치로 구체화한 명세입니다. 디자이너가 Figma 등에서 1:1로 옮길 수 있는 수준으로 작성했습니다.

### 3-1. 공통 레이아웃 스펙

기본 윈도우 크기는 1280×800px(최소 1024×720)입니다. 좌측 사이드바 폭은 고정 240px(배경 DeepNavy `#172554`), 우측 콘텐츠 영역은 가변이며 배경 OffWhite `#F8FAFC`입니다. 콘텐츠 영역 안쪽 여백은 상하좌우 32px입니다. 모든 카드/패널은 White `#FFFFFF`, 반경 12px, 그림자 `0 2px 8px rgba(15,23,42,0.08)`를 적용합니다.

사이드바 메뉴 항목은 높이 44px, 좌측 패딩 24px, 텍스트 14px입니다. 선택된 항목은 좌측에 4px 폭 SoliBlue `#1E40AF` 막대와 배경 `rgba(255,255,255,0.08)`로 강조합니다. 사이드바 상단 로고 영역은 높이 72px, 방패 아이콘 24px + "SoliGuard" 18px SemiBold(흰색)입니다.

### 3-2. 화면별 목업 스펙

**온보딩(직무 선택)** 화면은 사이드바 없이 전체 화면 중앙 정렬입니다. 상단에 방패 아이콘 48px와 "SoliGuard" Display 28px Bold, 그 아래 "내 PC의 고객 데이터, 먼저 찾습니다" Caption(Slate500). 중앙에 직무 카드 5개를 그리드(가로 3 + 2 배치, 카드 각 160×120px, 간격 16px)로 둡니다. 카드는 기본 Border `#E2E8F0` 1px, 선택 시 SoliBlue 2px 테두리 + 배경 `#EFF6FF`. 하단에 "모든 데이터는 이 PC 안에서만 처리됩니다" 체크 안내와 우하단 PrimaryButton "시작하기"(폭 120px, 높이 40px).

**메인 대시보드**는 사이드바 + 콘텐츠 구조입니다. 콘텐츠 상단에 위험 등급 카드(폭 100%, 높이 140px)를 두고, 좌측에 큰 원형 등급 표시(지름 96px, 안전 Green/주의 Amber/위험 Red 채움 + 등급 텍스트 Display 28px), 우측에 "마지막 점검 N일 전" Caption. 그 아래 "지금 점검하기" PrimaryButton을 폭 100%, 높이 64px, 폰트 18px로 가장 크게 배치하고 돋보기 아이콘 24px를 동반합니다. 버튼 하단에 "전체 스캔 / 빠른 스캔" 보조 텍스트 링크(SkyBlue). 맨 아래 "최근 활동" 리스트(각 행 높이 48px, 좌측 작은 컬러 점 + 텍스트 14px).

**스캔 진행** 화면은 중앙에 ProgressBar(폭 480px, 높이 12px, 반경 6px, 채움 SkyBlue, 트랙 Gray200)와 그 위 퍼센트 Display 28px. 아래에 현재 검사 파일 경로 Caption(Slate500, 말줄임 처리), "검사한 파일 N/M". 하단에 실시간 발견 집계를 2×2 그리드 카드(각 카드에 항목명 12px + 건수 22px Bold)로 보여줍니다.

**검출 결과** 화면은 상단 헤더(제목 H1 "점검 결과 — 위험 N건" + 우측 "전체 선택"/"리포트" 보조 버튼), 그 아래 결과 테이블입니다. 테이블 행 높이 56px, 컬럼은 체크박스(40px) / 위험도 태그(80px) / 파일·경로(가변) / 검출 항목(160px) / 조치 버튼(160px). 위험도 태그는 SeverityTag(높음 Red·중간 Amber·낮음 Green 채운 원 8px + 텍스트 12px). 행 hover 시 배경 `#F1F5F9`, 선택 시 `#EFF6FF`. 하단 고정 바(높이 64px)에 "선택한 N건에 대해:" + 마스킹/격리/완전삭제 버튼(완전삭제만 Danger 색). 우측 하단에 MaskedPreview 패널(폭 320px)을 두고, 검출값은 JetBrains Mono 14px로 `90****-2******` 형태만 표시.

**삭제 확인 모달(ConfirmDialog)**은 화면 중앙 모달(폭 480px), 상단에 ⚠ 아이콘(Amber/Red) + H2 "완전삭제 확인". 본문에 대상 파일 목록과 "되돌릴 수 없습니다" 경고(Slate900), "권장: 먼저 격리로 보관" 안내(Slate500). 하단에 텍스트 입력 필드("삭제" 입력 요구)와 버튼 3개: 취소(Border), 격리로 변경(SkyBlue), 삭제(Danger). 삭제 버튼은 정확히 "삭제"를 입력하기 전까지 disabled(회색).

**완료/리포트** 화면은 중앙에 ✅ 아이콘 48px(Green) + H1 "점검 완료". 처리 결과를 마스킹/격리/삭제 건수 3칸으로 보여주고, 위험 등급 변화를 "주의 🟡 → 안전 🟢" 형태로 화살표와 함께 표시. 그 아래 리포트 카드(폭 100%, 높이 100px)에 문서 아이콘 + "점검 진단서(PDF) 발급" + "리포트 저장" PrimaryButton. 맨 아래 "홈으로"/"격리함 보기" 보조 버튼.

### 3-3. 반응형·상태 표현 규칙

창 폭이 1024px 미만으로 줄면 사이드바를 아이콘만 남긴 폭 64px로 축소합니다. 버튼·행의 모든 인터랙션 상태(default/hover/pressed/disabled)는 명도 조절로 표현하되, hover는 8% 어둡게, pressed는 12% 어둡게, disabled는 투명도 40% + 회색조로 통일합니다. 로딩 상태는 SkyBlue 스피너 또는 ProgressBar로만 표현하고, 빈 상태(검출 0건)는 Green 체크 일러스트 + "안전합니다, 발견된 위험이 없습니다" 메시지로 안심을 줍니다.
